序列式容器

vector - 动态数组

#include <vector>
#include <iostream>

int main() {
    // 创建vector
    std::vector<int> vec = {1, 2, 3, 4, 5};
    
    // 添加元素
    vec.push_back(6);        // 末尾添加
    vec.insert(vec.begin() + 2, 10); // 指定位置插入
    
    // 访问元素
    std::cout << "第一个元素: " << vec.front() << std::endl;
    std::cout << "最后一个元素: " << vec.back() << std::endl;
    std::cout << "下标访问: " << vec[2] << std::endl;       // 不检查边界
    std::cout << "at()访问: " << vec.at(2) << std::endl;   // 检查边界
    
    // 遍历
    for (auto it = vec.begin(); it != vec.end(); ++it) {
        std::cout << *it << " ";
    }
    
    // 范围for循环 (C++11)
    for (int num : vec) {
        std::cout << num << " ";
    }
    
    // 删除元素
    vec.pop_back();                 // 删除末尾
    vec.erase(vec.begin() + 1);     // 删除指定位置
    vec.clear();                    // 清空
    
    // 容量操作
    std::cout << "大小: " << vec.size() << std::endl;
    std::cout << "容量: " << vec.capacity() << std::endl;
    vec.shrink_to_fit();            // 缩减容量
    
    return 0;
}
